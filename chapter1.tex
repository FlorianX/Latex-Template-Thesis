 & = \left.\frac{1}{2} x^2\right\vert_{a}^{b}\\
 & = \frac{1}{2} b^2 - \frac{1}{2} a^2 \\
 \intertext{mit $a=1$ und $b=3$ folgt:}
 \notag
 & = \frac{1}{2} \left(3^2 - 1^2\right)\\
 & = 5
\end{align}
